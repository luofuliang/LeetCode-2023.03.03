 *     Left *TreeNode
 *     Right *TreeNode

func bstToGst(root *TreeNode) *TreeNode {
    var sum int
    // 定义中序遍历函数，内部维护sum变量
    var traverse func(node *TreeNode)
    traverse = func(node *TreeNode) {
        if node == nil {
            return
        }
        traverse(node.Right)
        // 维护累加和
        sum += node.Val
        // 将BST转化为累加树
        node.Val = sum
        traverse(node.Left)
    }
    traverse(root)
    return root
}
